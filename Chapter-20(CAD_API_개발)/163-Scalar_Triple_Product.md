impl Vector3D {
    pub fn triple(a: &Vector3D, b: &Vector3D, c: &Vector3D) -> Real {
        let cx = b.y * c.z - b.z * c.y;
        let cy = b.z * c.x - b.x * c.z;
        let cz = b.x * c.y - b.y * c.x;
        a.x * cx + a.y * cy + a.z * cz
    }
}
